\termrow{$\mathcal{K}$}{The set of task kinds.}
\termrow{$\mathcal{I}$}{The set of task instances.}
\termrow{$\mathcal{A}$}{The set of task attempts.}
\termrow{$i \in \mathcal{I}$}{A task instance.}
\termrow{$a \in \mathcal{A}$}{A task attempt.}
\termrow{$k(i) \in \mathcal{K}$}{The task kind of task instance $i$.}
\termrow{$i(a) \in \mathcal{I}$}{The task instance executed by attempt $a$.}
\termrow{$k_a$}{A shorthand for $k(i(a))$.}

\termrow{$\mathcal{D}$}{The set of influence domains.}
\termrow{$D_{\mathrm{id}}$}{The task-identity influence domain.}
\termrow{$D_{\mathrm{intr}}$}{The intrinsic-feature influence domain.}
\termrow{$D_{\mathrm{ctx}}$}{The execution-context influence domain.}
\termrow{$D_{\mathrm{res}}$}{The resource-demand influence domain.}
\termrow{$D_{\mathrm{occ}}$}{The worker-occupancy influence domain.}
\termrow{$D_{\mathrm{sym}}$}{The symptom influence domain.}
\termrow{$D_{\mathrm{out}}$}{The outcome influence domain.}
\termrow{$D_{\mathrm{diag}}$}{The reliability-diagnostic influence domain.}
\termrow{$D_{\mathrm{risk}}$}{The operational-risk influence domain.}
\termrow{$D_{\mathrm{policy}}$}{The profiling-policy trigger influence domain.}
\termrow{$\delta_k:V_k\rightarrow\mathcal{D}$}{The domain-assignment function for the graph of task kind $k$.}
\termrow{$\delta(v)$}{The influence domain assigned to node $v$ when the task kind is clear from context.}
\termrow{$\leadsto_{\mathcal{D}}$}{The permitted domain-flow relation between influence domains.}

\termrow{$\mathcal{Z}_k$}{The canonical input-feature space for task kind $k$.}
\termrow{$z_a \in \mathcal{Z}_{k_a}$}{The canonical input-feature vector available for attempt $a$.}
\termrow{$\mathcal{X}_k$}{The intrinsic-feature space for task kind $k$.}
\termrow{$\mathcal{C}_k$}{The execution-context feature space for task kind $k$.}
\termrow{$P_X^k$}{The projection $P_X^k:\mathcal{Z}_k\rightarrow\mathcal{X}_k$.}
\termrow{$P_C^k$}{The projection $P_C^k:\mathcal{Z}_k\rightarrow\mathcal{C}_k$.}
\termrow{$\mathbf{x}_a$}{The intrinsic-feature projection $P_X^{k_a}(z_a)$.}
\termrow{$\mathbf{c}_a$}{The execution-context projection $P_C^{k_a}(z_a)$.}
\termrow{$\Theta_k$}{The model-state space for task kind $k$.}
\termrow{$\theta_k \in \Theta_k$}{Task-kind-specific model state.}

\termrow{$\mathcal{R}$}{The set of active resource-demand dimensions.}
\termrow{$\mathcal{O}$}{The set of worker-occupancy dimensions.}
\termrow{$\mathcal{Q}$}{The typed disjoint union $\mathcal{R}\dot{\cup}\mathcal{O}$.}
\termrow{$\mathcal{R}_k$}{The resource-demand dimensions relevant to task kind $k$.}
\termrow{$\mathcal{O}_k$}{The worker-occupancy dimensions relevant to task kind $k$.}
\termrow{$\mathcal{Q}_k$}{The cost dimensions relevant to task kind $k$, $\mathcal{R}_k\dot{\cup}\mathcal{O}_k$.}
\termrow{$q \in \mathcal{Q}$}{A typed cost dimension.}
\termrow{$\Delta_q$}{The descriptor for cost dimension $q$.}
\termrow{$\rho_q$}{The role of cost dimension $q$.}
\termrow{$u_q$}{The unit of cost dimension $q$.}
\termrow{$\mathcal{V}_q$}{The value domain of cost dimension $q$.}
\termrow{$\alpha_q$}{The attribution boundary for cost dimension $q$.}
\termrow{$\omega_q$}{The observation semantics for cost dimension $q$.}
\termrow{$\kappa_q$}{The composition operator for cost dimension $q$.}
\termrow{$\mathcal{E}_q$}{The estimate-object space for cost dimension $q$.}
\termrow{$\widehat{E}_q(a)$}{The estimate object for cost dimension $q$ during attempt $a$.}
\termrow{$\operatorname{point}_q$}{Projection from an estimate object to a point estimate.}
\termrow{$\operatorname{quantile}_{q,p}$}{Projection from an estimate object to a $p$-quantile estimate, when supported.}
\termrow{$\widehat{Y}_q(a)$}{The point estimate $\operatorname{point}_q(\widehat{E}_q(a))$.}
\termrow{$\widehat{Y}_{q,p}(a)$}{The estimated $p$-quantile for cost dimension $q$.}
\termrow{$Y_q(a)$}{The observed attempt-attributed value for cost dimension $q$, when available.}
\termrow{$\epsilon_q(a)$}{Residual error $Y_q(a)-\widehat{Y}_q(a)$, when $Y_q(a)$ is available.}
\termrow{$\widehat{\mathbf{D}}(a)$}{The resource-demand estimate-object vector for attempt $a$.}
\termrow{$\widehat{\mathbf{U}}(a)$}{The worker-occupancy estimate-object vector for attempt $a$.}
\termrow{$\widehat{\mathbf{C}}(a)$}{The structured attempt-level cost estimate $(\widehat{\mathbf{D}}(a),\widehat{\mathbf{U}}(a))$.}
\termrow{$F_k$}{The task-kind-specific cost estimator for task kind $k$.}
\termrow{$F_{k,q}$}{The dimension-specific estimator for task kind $k$ and dimension $q$.}
\termrow{$\widehat{B}_q$}{The estimated intrinsic baseline function for dimension $q$.}
\termrow{$\widehat{A}_q$}{The estimated contextual amplification factor for dimension $q$.}
\termrow{$\widehat{O}_q$}{The estimated additive overhead function for dimension $q$.}
\termrow{$\widehat{\mathbf{C}}_{\mathrm{base}}(a)$}{The cost estimate excluding additional profiling overhead.}
\termrow{$\widehat{\mathbf{C}}_{\rho}(a)$}{The cost estimate including overhead from profiling policy $\rho$.}

\termrow{$\mathcal{G}_k$}{The Task Behavior Graph associated with task kind $k$.}
\termrow{$V_k$}{The node set of $\mathcal{G}_k$.}
\termrow{$E_k$}{The edge set of $\mathcal{G}_k$.}
\termrow{$v \in V_k$}{A graph node.}
\termrow{$e \in E_k$}{A graph edge.}
\termrow{$\tau(v)$}{The node type assigned to node $v$.}
\termrow{$\lambda(e)$}{The edge role assigned to edge $e$.}
\termrow{$w_e$}{The estimated sensitivity, weight, or parameter associated with edge $e$.}
\termrow{$\gamma_e$}{The confidence value associated with edge $e$.}

\termrow{$\mathrm{Diag}(a)$}{Reliability diagnostics for attempt $a$.}
\termrow{$h_{\mathrm{diag}}$}{A diagnostic function that maps cost estimates, residuals, features, and model state to a diagnostic object.}
\termrow{$S(a)$}{Consequence information for attempt $a$, such as tenant criticality, dependency fragility, or queue-starvation impact.}
\termrow{$\mathrm{Risk}(a)$}{The operational-risk estimate for attempt $a$.}
\termrow{$g_{\mathrm{risk}}$}{A risk function that maps cost estimates, diagnostics, consequence information, and model state to operational risk.}
\termrow{$\mathcal{P}$}{The set of available profiling policies.}
\termrow{$B_{\mathrm{obs}}$}{The observation budget available to profiling-policy selection.}
\termrow{$J_p$}{A policy score or rule predicate for profiling policy $p$.}
\termrow{$\rho(a)$}{The profiling policy selected for attempt $a$ or a comparable class of attempts.}

\end{longtable}
